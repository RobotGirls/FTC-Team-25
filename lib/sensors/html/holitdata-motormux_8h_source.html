00128 /**
00129  * Read the status of the motors and tacho counts of the MMUX
00130  *
00131  * motorStatus is made of 3 bits, 1: motor is running, 0: motor is idle\n
00132  * bit 0: motor A\n
00133  * bit 1: motor B\n
00134  * bit 2: motor C\n
00135  *
00136  * @param link the MMUX port number
00137  * @param motorStatus status of the motors
00138  * @param tachoA Tacho count for motor A
00139  * @param tachoB Tacho count for motor B
00140  * @param tachoC Tacho count for motor C
00141  * @return true if no error occured, false if it did
00142  */
00143 bool HDMMUXreadStatus(tSensors link, ubyte &motorStatus, long &tachoA, long &tachoB, long &tachoC) {
00144   memset(HDMMUX_I2CRequest, 0, sizeof(tByteArray));
00145 
00146   HDMMUX_I2CRequest[0]  = 10;               // Message size
00147   HDMMUX_I2CRequest[1]  = HDMMUX_I2C_ADDR; // I2C Address
00148 
00149   if (!writeI2C(link, HDMMUX_I2CRequest, HDMMUX_I2CReply, 13))
00150     return false;
00151 
00152   motorStatus = HDMMUX_I2CReply[0];
00153 
00154   // Assemble and assign the encoder values
00155   tachoA = (HDMMUX_I2CReply[1] << 24) + (HDMMUX_I2CReply[2] << 16) + (HDMMUX_I2CReply[3] << 8) + (HDMMUX_I2CReply[4] << 0);
00156   tachoB = (HDMMUX_I2CReply[5] << 24) + (HDMMUX_I2CReply[6] << 16) + (HDMMUX_I2CReply[7] << 8) + (HDMMUX_I2CReply[8] << 0);
00157   tachoC = (HDMMUX_I2CReply[9] << 24) + (HDMMUX_I2CReply[10] << 16) + (HDMMUX_I2CReply[11] << 8) + (HDMMUX_I2CReply[12] << 0);
00158 
00159   return true;
00160 }

00163 /**
00164  * Send a command to the MMUX.
00165  *
00166  * Note: this is an internal function and shouldn't be used directly
00167  * @param link the MMUX port number
00168  * @param mode the mode the MMX should operate in, controlling motors, resetting tachos or settings new address
00169  * @param channel the motors the command applies to
00170  * @param rotparams the additional parameters that make up the command
00171  * @param duration the number of units (can be seconds, rotations or degrees) the command should be run for
00172  * @param power the amount of power to be applied to the motor(s)
00173  * @param steering used for syncronised movement to control the amount of steering
00174  * @return true if no error occured, false if it did
00175  */
00176 bool HDMMUXsendCommand(tSensors link, ubyte mode, ubyte channel, ubyte rotparams, long duration, byte power, byte steering) {
00177   memset(HDMMUX_I2CRequest, 0, sizeof(tByteArray));
00178 
00179   HDMMUX_I2CRequest[0]  = 10;               // Message size
00180   HDMMUX_I2CRequest[1]  = HDMMUX_I2C_ADDR; // I2C Address
00181   HDMMUX_I2CRequest[2]  = mode;
00182   HDMMUX_I2CRequest[3]  = channel;
00183   HDMMUX_I2CRequest[4]  = rotparams;
00184   HDMMUX_I2CRequest[5]  = (duration >> 24) & 0xFF;
00185   HDMMUX_I2CRequest[6]  = (duration >> 16) & 0xFF;
00186   HDMMUX_I2CRequest[7]  = (duration >>  8) & 0xFF;
00187   HDMMUX_I2CRequest[8]  = (duration >>  0) & 0xFF;
00188   HDMMUX_I2CRequest[9]  = power;
00189   HDMMUX_I2CRequest[10] = (byte)(steering & 0xFF);
00190 
00191   return writeI2C(link, HDMMUX_I2CRequest);
00192 }

00195 /**
00196  * Run motor with specified speed.
00197  *
00198  * @param muxmotor the motor-MUX motor
00199  * @param power power the amount of power to apply to the motor, value between -100 and +100
00200  * @return true if no error occured, false if it did
00201  */
00202 bool HDMMotor(tMUXmotor muxmotor, byte power) {
00203   ubyte command = 0;
00204   bool retval = true;
00205   long target = mmuxData[SPORT(muxmotor)].target[MPORT(muxmotor)];
00206 
00207   command |= (mmuxData[SPORT(muxmotor)].brake[MPORT(muxmotor)]) ? HDMMUX_ROT_BRAKE : HDMMUX_ROT_FLOAT;
00208   command |= mmuxData[SPORT(muxmotor)].ramping[MPORT(muxmotor)];
00209   command |= (mmuxData[SPORT(muxmotor)].pidcontrol[MPORT(muxmotor)]) ? HDMMUX_ROT_POWERCONTROL : 0;
00210   command |= (power > 0) ? HDMMUX_ROT_FORWARD : HDMMUX_ROT_REVERSE;
00211   command |= mmuxData[SPORT(muxmotor)].targetUnit[MPORT(muxmotor)];
00212 
00213   retval = HDMMUXsendCommand((tSensors)SPORT(muxmotor), HDMMUX_CMD_MOTOR, (ubyte)MPORT(muxmotor) + 1, command, target, abs(power), 0);
00214 
00215   // Reset the data
00216   mmuxData[SPORT(muxmotor)].targetUnit[MPORT(muxmotor)] = HDMMUX_ROT_UNLIMITED;
00217   mmuxData[SPORT(muxmotor)].target[MPORT(muxmotor)] = 0;
00218 
00219   return retval;
00220 }
