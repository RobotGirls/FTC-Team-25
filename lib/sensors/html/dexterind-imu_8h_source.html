00102 /**
00103  * Configure the gyro
00104  * @param link the port number
00105  * @param range the operating range of the gyro
00106  * @param lpfenable Enable built-in Low Pass Filter to reduce spikes in data, optional, defaults to true.
00107  * @return true if no error occured, false if it did
00108  */
00109 bool DIMUconfigGyro(tSensors link, ubyte range, bool lpfenable){
00110         memset(DIMU_I2CRequest, 0, sizeof(DIMU_I2CRequest));
00111 
00112         // Setup the size and address, same for all requests.
00113         DIMU_I2CRequest[0] = 3;    // Sending address, register, value. Optional, defaults to true
00114         DIMU_I2CRequest[1] = 0xD2; // I2C Address of gyro.
00115 
00116         // Write CTRL_REG2
00117         // No High Pass Filter
00118         DIMU_I2CRequest[2] = DIMU_GYRO_CTRL_REG2;
00119         DIMU_I2CRequest[3] = 0x00;
00120         if (!writeI2C(link, DIMU_I2CRequest))
00121           return false;
00122 
00123         // Write CTRL_REG3
00124         // No interrupts.  Date ready.
00125         ////////////////////////////////////////////////////////////////////////////
00126         DIMU_I2CRequest[2] = DIMU_GYRO_CTRL_REG3;      // Register address of CTRL_REG3
00127         DIMU_I2CRequest[3] = 0x08;      // No interrupts.  Date ready.
00128         if(!writeI2C(link, DIMU_I2CRequest))
00129           return false;
00130 
00131         // Write CTRL_REG4
00132         // Full scale range.
00133         DIMU_I2CRequest[2] = DIMU_GYRO_CTRL_REG4;
00134         DIMU_I2CRequest[3] = range + DIMU_CTRL4_BLOCKDATA;
00135         writeI2C(link, DIMU_I2CRequest);
00136 
00137         //Write CTRL_REG5
00138         DIMU_I2CRequest[2] = DIMU_GYRO_CTRL_REG5;      // Register address of CTRL_REG5
00139         DIMU_I2CRequest[3] = (lpfenable) ? 0x02 : 0x00;      // filtering - low pass
00140         if (!writeI2C(link, DIMU_I2CRequest))
00141           return false;
00142 
00143         // Write CTRL_REG1
00144         // Enable all axes. Disable power down.
00145         DIMU_I2CRequest[2] = DIMU_GYRO_CTRL_REG1;
00146         DIMU_I2CRequest[3] = 0x0F;
00147         if (!writeI2C(link, DIMU_I2CRequest))
00148           return false;
00149 
00150         // Set DIMU_Gyro_divisor so that the output of our gyro axis readings can be turned
00151         // into scaled values.
00152         ///////////////////////////////////////////////////////////////////////////
00153         if(range == 0)
00154           DIMU_Gyro_divisor[link] = 128;      // Full scale range is 250 dps.
00155   else if (range == 0x10)
00156     DIMU_Gyro_divisor[link] = 64;       // Full scale range is 500 dps.
00157         else if (range == 0x30)
00158           DIMU_Gyro_divisor[link] = 16;       // Full scale range is 2000 dps.
00159 
00160         return true;
00161 }

00212  * Wait for the I2C bus to be ready for the next message
00213  * @param link the port number
00214  * @param range the range at which to operate the Accelerometer, can be 2, 4 and 8G
00215  * @return true if no error occured, false if it did

00217 bool DIMUconfigAccel(tSensors link, ubyte range) {
00218   switch(range) {
00219     case DIMU_ACC_RANGE_2G: DIMU_Accel_divisor[link] = 64.0; break;
00220     case DIMU_ACC_RANGE_4G: DIMU_Accel_divisor[link] = 32.0; break;

00223 
00224         DIMU_I2CRequest[0] = 3;                 // Sending address, register, value.
00225         DIMU_I2CRequest[1] = DIMU_ACC_I2C_ADDR; // I2C Address of Accelerometer.
00226 
00227         //Set the Mode Control - P.25 of Documentation
00228         ////////////////////////////////////////////////////////////////////////////
00229         DIMU_I2CRequest[2] = 0x16;                   // Register address of Mode Control
00230         DIMU_I2CRequest[3] = range | DIMU_ACC_MODE_MEAS;
00231         if (!writeI2C(link, DIMU_I2CRequest[0]))     // (Port 1, Message Array, Reply Size)
00232           return false;
00233 
00234         DIMUcalAccel(link);

00247         DIMU_I2CRequest[0] = 2;      // Sending address, register.

00287 /**
00288  * Read the specified accelerometer axis, returns an 10 bit answer
00289  * @param link the port number
00290  * @param axis the specific axis
00291  * @param calibrate optional argument, if set to to true, the sensor will calibrate this axis first.
00292  * @return gravity in G with 10 bit accuracy
00293  */
00294 float DIMUreadAccelAxis10Bit(tSensors link, ubyte axis, bool calibrate){
00295   int ureading = 0;  // unsigned sensor data
00296   int sreading = 0;  // signed sensor data
00297   int drift_offset = 0;
00298 
00299   if (calibrate == true) {
00300     writeDebugStreamLine("axis: %d", axis);
00301     DIMUsetAccelAxisOffset(link, 0x10 + axis, 0x00, 0x00);
00302     wait1Msec(50);
00303   }
00304 
00305   DIMU_I2CRequest[0] = 2;          // Sending address, register.
00306   DIMU_I2CRequest[1] = DIMU_ACC_I2C_ADDR;  // I2C Address of accl.
00307   DIMU_I2CRequest[2] = axis;        // First Register of the data we're requesting.
00308 
00309   if (!writeI2C(link, DIMU_I2CRequest[0], DIMU_I2CReply, 2))
00310     return 0;
00311 
00312   ureading = DIMU_I2CReply[0] + (DIMU_I2CReply[1] << 8) & 0x3FF;
00313   sreading = (ureading > 511) ? ureading - 1024 : ureading;
00314   //sreading = (ureading & 0x200) ? -(((~ureading) & 0x3FF)+1) : ureading;
00315 
00316   if (calibrate == true) {
00317     wait1Msec(50);
00318     switch (axis) {
00319       case DIMU_ACC_X_AXIS: drift_offset = (  0 - sreading ) * 2; break;
00320       case DIMU_ACC_Y_AXIS: drift_offset = (  0 - sreading ) * 2; break;
00321       case DIMU_ACC_Z_AXIS: drift_offset = ( 64 - sreading ) * 2; break;
00322     }
00323     DIMUsetAccelAxisOffset(link, 0x10 + axis, drift_offset & 0x00ff, (drift_offset & 0xff00 ) >> 8);
00324   }
00325 
00326   return sreading / 64.0;
00327 }
